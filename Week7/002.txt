#include <iostream>
#include <string>
using namespace std;

//#pragma warning(disable:4996)
//#define cin std::ios::sync_with_stdio(false);cin
int table[128] = { 0 };

int main() {
	//97～122号为26个小写英文字母
	string s;
	getline(cin, s);
	for (int i = 0; i < s.length(); i++) {
		table[s[i]]++;
	}
	for (int i = 0; i < s.length(); i++) {
		if (table[s[i]] == 1) {
			cout << char(s[i]);
			return 0;
		}
	}
	cout << "no";
	return 0;
}
