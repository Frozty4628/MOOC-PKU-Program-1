#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

//#pragma warning(disable:4996)

int bitManipulation2(int n, int i);

int main() {
	int t, n, i;
	cin >> t;
	while (t--) {
		cin >> n >> i;
		cout << bitManipulation2(n, i) << endl;
	}
	return 0;
}

int bitManipulation2(int n, int i) {
	return n ^ (1 << i);
	return 0;
}
